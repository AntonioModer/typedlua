\section{Type Language}

\begin{align*}
T ::= \; & C \; | \; B \; | \; Object \; | \; Any\\
C ::= \; & \mathbf{nil} \; | \; \mathbf{false} \; | \; \mathbf{true} \;
 | \; {<}double{>} \; | \; {<}integer{>} \; | \; {<}string{>}\\
B ::= \; & Boolean \; | \; Number \; | \; String\\
\end{align*}
